  const data = createProjectDependencySchema.parse(input);
  const user = await requireUser();

  return withTxRetry(() => db.$transaction(async (tx) => {
    // Sorted pair for advisory lock ŌĆö same lock for AŌåÆB as BŌåÆA.
    const [low, high] = [data.dependentProjectId, data.dependsOnProjectId].sort();
    await tx.$executeRawUnsafe(
      `SELECT pg_advisory_xact_lock(hashtext($1), hashtext($2))`,
      low, high,
    );

    // Cycle detection via recursive CTE.
    // Would inserting (dep, dependsOn) create a cycle?
    // A cycle exists if dependsOn already transitively depends on dep.
    const cycle = await tx.$queryRawUnsafe<Array<{ exists: boolean }>>(
      `WITH RECURSIVE descendants AS (
        SELECT "dependsOnProjectId" AS pid FROM "ProjectDependency"
          WHERE "dependentProjectId" = $1
        UNION
        SELECT pd."dependsOnProjectId" FROM "ProjectDependency" pd
          INNER JOIN descendants d ON pd."dependentProjectId" = d.pid
      )
      SELECT EXISTS (SELECT 1 FROM descendants WHERE pid = $2) AS exists`,
      data.dependsOnProjectId, data.dependentProjectId,
    );
    if (cycle[0]?.exists) {
      throw new Error("Edge would create a cycle in the dependency graph.");
    }

    return tx.projectDependency.create({
      data: { ...data, createdById: user.id },
    });
  }, { isolationLevel: "Serializable" }));
}

export async function editProjectDependency(input: unknown) {
  const { id, ...data } = editProjectDependencySchema.parse(input);
  await requireUser();
  return db.projectDependency.update({ where: { id }, data });
}

export async function deleteProjectDependency(id: string) {
  await requireUser();
  await db.projectDependency.delete({ where: { id } });
}

import type { Stage } from "@prisma/client";
import type { PrismaTx } from "@/lib/db-types";
import type { GateResult } from "@/lib/stages";

const STAGE_ORDER: Stage[] = ["REQUIREMENTS","SCHEMATIC","BOM_SOURCING","LAYOUT","DRC_GERBER","ORDERING","ASSEMBLY","BRINGUP","REVISION"];

function stageIndex(s: Stage): number {
  return STAGE_ORDER.indexOf(s);
}

export async function checkProjectDependencies(
  tx: PrismaTx,
  projectId: string,
  currentStage: Stage,
): Promise<GateResult> {
  // Pull all outbound edges where the gate would be active given currentStage.
  const edges = await tx.projectDependency.findMany({
    where: { dependentProjectId: projectId },
    include: {
      dependsOnProject: {
        include: {
          revisions: {
            orderBy: { createdAt: "desc" },
            take: 1,
            select: { currentStage: true, label: true },
          },
        },
      },
    },
  });

  const reasons: string[] = [];
  for (const edge of edges) {
    if (stageIndex(currentStage) < stageIndex(edge.dependentStageGated)) continue;
    const depRev = edge.dependsOnProject.revisions[0];
    if (!depRev) {
      reasons.push(`Depends on project ${edge.dependsOnProject.slug ?? edge.dependsOnProject.id} at ${edge.dependsOnStageRequired}; it has no revisions.`);
      continue;
    }
    if (stageIndex(depRev.currentStage) < stageIndex(edge.dependsOnStageRequired)) {
      reasons.push(`Depends on ${edge.dependsOnProject.slug ?? edge.dependsOnProject.id} at ${edge.dependsOnStageRequired}; latest revision is at ${depRev.currentStage}.`);
    }
  }

  return reasons.length === 0 ? { ok: true } : { ok: false, reasons };
}

import type { Stage } from "@prisma/client";
import type { PrismaTx } from "@/lib/db-types";

const STAGE_ORDER: Stage[] = ["REQUIREMENTS","SCHEMATIC","BOM_SOURCING","LAYOUT","DRC_GERBER","ORDERING","ASSEMBLY","BRINGUP","REVISION"];

function stageIndex(s: Stage): number {
  return STAGE_ORDER.indexOf(s);
}

export async function dependentsAtRisk(
  tx: PrismaTx,
  projectId: string,
  fromStage: Stage,
  toStage: Stage,
) {
  // Find all edges where I'm the dependency AND the required stage is being regressed past.
  const fromIdx = stageIndex(fromStage);
  const toIdx = stageIndex(toStage);
  if (toIdx >= fromIdx) return [];

  return tx.projectDependency.findMany({
    where: { dependsOnProjectId: projectId },
    include: { dependentProject: { select: { slug: true, name: true } } },
  }).then((edges) =>
    edges.filter((e) => {
      const requiredIdx = stageIndex(e.dependsOnStageRequired);
      // The dependency was satisfying e at fromStage >= requiredIdx; after regress to toStage,
      // it's at risk iff toStage < requiredIdx.
      return fromIdx >= requiredIdx && toIdx < requiredIdx;
    }),
  );
}
